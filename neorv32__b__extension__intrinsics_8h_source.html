657 uint32_t riscv_emulate_clz(uint32_t rs1) {
658 
659  uint32_t sreg = rs1;
660  uint32_t cnt = 0;
661 
662  while(1) {
663  if (sreg & 0x80000000UL) {
664  break;
665  }
666  else {
667  sreg <<= 1;
668  cnt++;
669  }
670  }
671 
672  return cnt;
673 }

936
