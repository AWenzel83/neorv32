86 float subnormal_flush(float tmp) {
87 
88  float res = tmp;
89 
90  if (fpclassify(tmp) == FP_SUBNORMAL) {
91  if (signbit(tmp) != 0) {
92  res = -0.0f;
93  }
94  else {
95  res = +0.0f;
96  }
97  }
98 
99  return res;
100 }

129 uint32_t get_sw_exceptions(void) {
130 
131  const uint32_t FP_EXC_NV_C = 1 << 0; // invalid operation
132  const uint32_t FP_EXC_DZ_C = 1 << 1; // divide by zero
133  const uint32_t FP_EXC_OF_C = 1 << 2; // overflow
134  const uint32_t FP_EXC_UF_C = 1 << 3; // underflow
135  const uint32_t FP_EXC_NX_C = 1 << 4; // inexact
136 
137  int fpeRaised = fetestexcept(FE_ALL_EXCEPT);
138 
139  uint32_t res = 0;
140 
141  if (fpeRaised & FE_INVALID) { res |= FP_EXC_NV_C; }
142  if (fpeRaised & FE_DIVBYZERO) { res |= FP_EXC_DZ_C; }
143  if (fpeRaised & FE_OVERFLOW) { res |= FP_EXC_OF_C; }
144  if (fpeRaised & FE_UNDERFLOW) { res |= FP_EXC_UF_C; }
145  if (fpeRaised & FE_INEXACT) { res |= FP_EXC_NX_C; }
146 
147  feclearexcept(FE_ALL_EXCEPT);
148 
149  return res;
150 }

167 
168  register uint32_t result __asm__ ("a0");
169  register uint32_t tmp_a __asm__ ("a0") = rs1;
170  register uint32_t tmp_b __asm__ ("a1") = rs2;
171 
172  // dummy instruction to prevent GCC "constprop" optimization

174 
175  // fadd.s a0, a0, a1
176  CUSTOM_INSTR_R2_TYPE(0b0000000, a1, a0, 0b000, a0, 0b1010011);
177 
178  return result;

768 /**********************************************************************/
775 float riscv_emulate_fmins(float rs1, float rs2) {
776 
777  union {
778  uint32_t binary_value;
779  float float_value;
780  } tmp_a, tmp_b;
781 
782  if ((fpclassify(rs1) == FP_NAN) && (fpclassify(rs2) == FP_NAN)) {
783  return nanf("");
784  }
785 
786  if (fpclassify(rs1) == FP_NAN) {
787  return rs2;
788  }
789 
790  if (fpclassify(rs2) == FP_NAN) {
791  return rs1;
792  }
793 
794  // RISC-V spec: -0 < +0
795  tmp_a.float_value = rs1;
796  tmp_b.float_value = rs2;
797  if (((tmp_a.binary_value == 0x80000000) && (tmp_b.binary_value == 0x00000000)) ||
798  ((tmp_a.binary_value == 0x00000000) && (tmp_b.binary_value == 0x80000000))) {
799  return -0.0f;
800  }
801 
802  return subnormal_flush(fmin(rs1, rs2));
